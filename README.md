### API Endpoint

```http
GET https://api.basebot.md/status
```

### Response Format

```json
{
  "status": "online",
  "version": "2.0.0",
  "users": 1200,
  "groups": 45,
  "commands": 150,
  "plugins": 23,
  "uptime": "99.9%",
  "ai_status": "operational",
  "timestamp": "2026-04-07T12:00:00Z",
  "system": {
    "memory": "128MB",
    "cpu": "12%",
    "node_version": "24.x"
  }
}
```

### Node.js Integration

```javascript
// Add to your bot's status endpoint
const express = require('express');
const app = express();

app.get('/api/status', (req, res) => {
  const status = {
    status: global.botConnected ? 'online' : 'offline',
    version: require('./package.json').version,
    users: Object.keys(global.db.users).length,
    groups: Object.keys(global.db.groups).length,
    commands: global.commands.size,
    plugins: global.plugins.length,
    uptime: process.uptime(),
    ai_status: global.aiCore?.status || 'disabled',
    timestamp: new Date().toISOString(),
    system: {
      memory: `${Math.round(process.memoryUsage().heapUsed / 1024 / 1024)}MB`,
      cpu: process.cpuUsage(),
      node_version: process.version
    }
  };
  res.json(status);
});

app.listen(3000, () => console.log('Status API running on port 3000'));
```
